📋 Setup Guide — How to make this fully live (free tools only)

Step 1 — Google Sheets backend (free)

  1. Go to sheets.google.com and create a new spreadsheet
  2. Name the first sheet Registrations
  3. Add headers in Row 1: Timestamp | FirstName | LastName | Age | Gender | Email | Contact | Date | CouponCode | Used
  4. Go to Extensions → Apps Script
  5. Paste the Google Apps Script code (in the second details block below)
  6. Click Deploy → New Deployment → Web App
  7. Set "Who has access" to Anyone, click Deploy
  8. Copy the Web App URL and paste it into this HTML file replacing YOUR_GOOGLE_SCRIPT_URL

Step 2 — Brevo for emails (free, 300 emails/day = 1,200 over 4 days)

  1. Sign up free at brevo.com — the free plan gives 300 emails/day, covering all 400 registrations across 4 days
  2. Go to Settings → API Keys and click Generate a new API key. Copy it.
  3. Go to Senders & Domains → Senders and add + verify the email you want to send from (e.g. hello@crvngjunction.com)
  4. Open your Google Apps Script (from Step 1) and paste your API key and sender email into the two lines at the top: BREVO_API_KEY and BREVO_SENDER_EMAIL
  5. ⚠ Important: The API key lives only inside Apps Script — it is never in the HTML file. This keeps it private and avoids the GitHub secret scanning warning.
  6. No template setup needed — the email HTML is built directly into the Apps Script code

Step 3 — Host on GitHub Pages (free, permanent URL)

  1. Create a free account at github.com
  2. Create a new repository, name it crvng-junction
  3. Upload this HTML file, name it index.html
  4. Go to Settings → Pages → Source: Deploy from branch (main)
  5. Your URL will be: https://yourusername.github.io/crvng-junction

Step 4 — Generate QR code (free)

  1. Go to qr-code-generator.com
  2. Paste your GitHub Pages URL
  3. Download and print the QR code

const SHEET_NAME = 'Registrations';
const MAX_PER_DAY = 100;
const OFFER_DATES = ['2025-05-21','2025-05-22','2025-05-23','2025-05-24'];

// PUT YOUR BREVO DETAILS HERE — this file is private, never exposed publicly
const BREVO_API_KEY = 'YOUR_BREVO_API_KEY';
const BREVO_SENDER_EMAIL = 'YOUR_VERIFIED_SENDER_EMAIL';
const BREVO_SENDER_NAME = 'Crvng Junction';

function doGet(e) {
  const action = e.parameter.action;
  if (action === 'getAvailability') return getAvailability();
  if (action === 'validate') return validateCoupon(e.parameter.code);
  return ContentService.createTextOutput('OK');
}

function doPost(e) {
  const data = JSON.parse(e.postData.contents);
  if (data.action === 'register') return register(data);
  return jsonResponse({success:false, error:'Unknown action'});
}

function getAvailability() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  const rows = sheet.getDataRange().getValues();
  const counts = {};
  OFFER_DATES.forEach(d => counts[d] = 0);
  rows.slice(1).forEach(r => { if (counts[r[7]] !== undefined) counts[r[7]]++; });
  const avail = {};
  OFFER_DATES.forEach(d => avail[d] = Math.max(0, MAX_PER_DAY - (counts[d]||0)));
  return jsonResponse({success:true, availability: avail});
}

function register(data) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  const rows = sheet.getDataRange().getValues();
  const dateCount = rows.slice(1).filter(r => r[7] === data.date).length;
  if (dateCount >= MAX_PER_DAY) return jsonResponse({success:false, error:'FULL'});
  const existing = rows.slice(1).find(r => r[5] === data.email && r[7] === data.date);
  if (existing) return jsonResponse({success:false, error:'DUPLICATE'});
  const code = 'CRVNG-' + Math.random().toString(36).substring(2,8).toUpperCase();
  sheet.appendRow([new Date(),data.firstName,data.lastName,data.age,data.gender,data.email,data.contact,data.date,code,'false']);
  sendBrevoEmail(data.email, data.firstName, code, data.date);
  return jsonResponse({success:true, couponCode: code});
}

function validateCoupon(code) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  const rows = sheet.getDataRange().getValues();
  const idx = rows.findIndex(r => r[8] === code);
  if (idx < 1) return jsonResponse({success:true, valid:false, reason:'Code not found'});
  const row = rows[idx];
  if (row[9] === 'true') return jsonResponse({success:true, valid:false, reason:'Already redeemed', name: row[1]+' '+row[2], date: row[7]});
  sheet.getRange(idx+1, 10).setValue('true');
  return jsonResponse({success:true, valid:true, name: row[1]+' '+row[2], date: row[7], email: row[5]});
}

function jsonResponse(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON);
}
